import { TimelineWrapper } from '@/components/TimelineWrapper'
import { timelineEntries } from '@/components/timeline-data'
import { TimelineEntry } from '@/components/TimelineEntry'

export const metadata = {
  title: 'Whoami',
}

export const dynamic = 'force-static'

<div className="content-width mb-8 mobile:mb-10">
  <h1 className="text-page-title font-bold text-rurikon-700 page-title-gap tracking-tight eb-garamond-italic">Whoami</h1>
  <p className="text-rurikon-400 text-section-lede eb-garamond-italic">Security researcher and software engineer. I make and break things professionally (or for fun, legally of course).</p>
</div>

<TimelineWrapper initialYear={timelineEntries[timelineEntries.length - 1].year}>
  {(() => {
    const BASE_DELAY = 0.12
    const DELAY_STEP = 0.35
    const reversedEntries = [...timelineEntries].reverse()

    return reversedEntries.map((entry, index) => (
      <TimelineEntry
        key={`${entry.year}-${entry.month}-${index}`}
        entry={entry}
        delay={BASE_DELAY + index * DELAY_STEP}
      />
    ))
  })()}
</TimelineWrapper>
